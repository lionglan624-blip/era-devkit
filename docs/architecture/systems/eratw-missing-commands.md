### 2.5 射精位置系（体位連動システム）

**設計方針**: 各体位にデフォルト射精位置を設定し、オプションで変更可能にする。

#### 2.5.1 射精位置定義

| 射精位置ID | 名称 | 汚れ部位 |
|:----------:|------|:--------:|
| 1 | 膣内 | Ｖ |
| 2 | アナル内 | Ａ |
| 3 | 口内 | 口 |
| 4 | 顔 | 顔 |
| 5 | 胸 | 胸 |
| 6 | 腹 | 体 |
| 7 | 尻 | 尻 |
| 8 | 髪 | 髪 |

#### 2.5.2 体位別デフォルト射精位置

; ========================================
; 体位別デフォルト射精位置テーブル
; ========================================

@GET_DEFAULT_EJACULATION(体位)
  SELECTCASE 体位
    ; ========== 膣挿入系 → デフォルト: 膣内 ==========
    CASE 60, 61, 64, 65, 67, 68, 73, 74, 75, 76, 77, 78

    CASE 62, 63, 66, 69, 70, 92, 93, 94
      ; 正常位アナル, 後背位アナル, 騎乗位アナル, 座位アナル, 逆アナル系
      RESULT = 2  ; アナル内

    ; ========== 口淫系 → デフォルト: 顔 ==========
    CASE 81, 140, 511
      ; フェラチオ, イラマチオ, 二本フェラ
      RESULT = 4  ; 顔

    ; ========== パイズリ系 → デフォルト: 胸/顔 ==========
    CASE 82, 508, 512
      ; パイズリ, 69パイズリ, パイズリフェラ
      RESULT = 5  ; 胸

    ; ========== 素股系 → デフォルト: 腹 ==========
    CASE 83, 510
      ; 素股, ダブル素股
      RESULT = 6  ; 腹

    ; ========== 手淫系 → デフォルト: 腹/体 ==========
    CASE 80, 505
      ; 手淫, 手淫フェラ
      RESULT = 6  ; 腹

    ; ========== 足コキ系 → デフォルト: 足/体 ==========
    CASE 85, 123
      ; 足コキ, 足コキする
      RESULT = 9  ; 外出し

    ; ========== 69系 → デフォルト: 口内 ==========
    CASE 501
      ; シックスナイン
      RESULT = 3  ; 口内

    ; ========== その他 → デフォルト: 外出し ==========
    CASEELSE
      RESULT = 9  ; 外出し
  ENDSELECT

; ========================================
; 射精位置の選択肢表示（体位による制限）
; ========================================

@GET_AVAILABLE_EJACULATION_OPTIONS(体位)

      SETBIT オプション, 1  ; 膣内
      SETBIT オプション, 6  ; 腹
      SETBIT オプション, 7  ; 尻
      SETBIT オプション, 9  ; 外出し

    ; アナル挿入系: アナル内, 外出し(腹/尻)
    CASE 62, 63, 66, 69, 70, 92, 93, 94
      SETBIT オプション, 2  ; アナル内
      SETBIT オプション, 6  ; 腹
      SETBIT オプション, 7  ; 尻
      SETBIT オプション, 9  ; 外出し

    ; 口淫系: 口内, 顔, 髪
    CASE 81, 140, 511
      SETBIT オプション, 3  ; 口内
      SETBIT オプション, 4  ; 顔
      SETBIT オプション, 8  ; 髪

#### 2.5.3 射精時の処理フロー

```erb
; ========================================
; 射精処理（体位連動）
; ========================================

@PROCESS_EJACULATION(射精者, 体位)
  ; デフォルト位置を取得
  LOCAL デフォルト = CALL GET_DEFAULT_EJACULATION(体位)

  ; 設定で「射精位置選択を表示」がONの場合のみ選択肢表示
  IF FLAG:射精位置選択表示
    LOCAL 選択肢 = CALL GET_AVAILABLE_EJACULATION_OPTIONS(体位)
    LOCAL 選択 = CALL EJACULATION_SELECT_UI(デフォルト, 選択肢)
    IF 選択 != -1
      デフォルト = 選択
    ENDIF
  ENDIF

  ; 射精位置を適用
  CALL APPLY_EJACULATION(射精者, デフォルト)
  RETURN

@EJACULATION_SELECT_UI(デフォルト, 選択肢ビット)
  ; 射精直前に表示される選択UI
  PRINTL ──射精位置──

  LOCAL 番号 = 1
  LOCAL 選択マップ:10  ; 表示番号→位置IDのマッピング

  ; デフォルトを最初に表示
  PRINTFORML [0] %CALL GET_EJACULATION_NAME(デフォルト)%（このまま）

  ; 他の選択肢を表示
  FOR 位置, 1, 9
    SIF 位置 == デフォルト
      CONTINUE
    IF GETBIT(選択肢ビット, 位置)
      PRINTFORML [{番号}] %CALL GET_EJACULATION_NAME(位置)%
      選択マップ:番号 = 位置
      番号++
    ENDIF
  NEXT

  INPUT 入力

  IF 入力 == 0
    RESULT = -1  ; デフォルトのまま
  ELSE
    RESULT = 選択マップ:入力
  ENDIF
  RETURN

@GET_EJACULATION_NAME(位置ID)
  SELECTCASE 位置ID
    CASE 1
      RESULTS = "中出し"
    CASE 2
      RESULTS = "アナル内"
    CASE 3
      RESULTS = "口内射精"
    CASE 4
      RESULTS = "顔射"
    CASE 5
      RESULTS = "胸に"
    CASE 6
      RESULTS = "腹に"
    CASE 7
      RESULTS = "尻に"
    CASE 8
      RESULTS = "髪に"
    CASE 9
      RESULTS = "外出し"
  ENDSELECT
  RETURN

@APPLY_EJACULATION(射精者, 位置ID)
  ; 射精位置に応じた処理

  ; TCVAR設定
  SELECTCASE 位置ID
    CASE 1
      TCVAR:射精者:2 = 1  ; 膣内
    CASE 2
      TCVAR:射精者:2 = 2  ; アナル
    CASE 3, 4, 8
      TCVAR:射精者:2 = 4  ; 口淫扱い
    CASE 5, 6, 7, 9
      TCVAR:射精者:2 = 8  ; 体表
  ENDSELECT

  ; 汚れ処理
  SELECTCASE 位置ID
    CASE 1
      STAIN:TARGET:Ｖ |= BIT:精液
      ; 妊娠判定へ
      CALL 妊娠判定(TARGET, 射精者)
    CASE 2
      STAIN:TARGET:Ａ |= BIT:精液
    CASE 3
      STAIN:TARGET:口 |= BIT:精液
      ; 精液嚥下判定
      IF RAND:100 < 50 + ABL:TARGET:精液中毒 * 10
        PRINTL %CALLNAME:TARGET%は精液を飲み込んだ
      ENDIF
    CASE 4
      STAIN:TARGET:顔 |= BIT:精液
    CASE 5
      STAIN:TARGET:胸 |= BIT:精液
    CASE 6
      STAIN:TARGET:体 |= BIT:精液
    CASE 7
      STAIN:TARGET:尻 |= BIT:精液
    CASE 8
      STAIN:TARGET:髪 |= BIT:精液
    CASE 9
      STAIN:TARGET:体 |= BIT:精液
  ENDSELECT

  ; 口上呼び出し
  CALL KOJO_EJACULATION_LOCATION(TARGET, 射精者, 位置ID)
  RETURN
```

### 3.4 輪姦システム（新規）

```erb
; ========================================
; 輪姦システム
; ========================================

; 輪姦開始
@GANGBANG_START(被害者, 加害者リスト, 人数)
  #DIM 被害者
  #DIM 加害者リスト:10
  #DIM 人数

  ; 輪姦状態セット
  TFLAG:輪姦中 = 1
  TFLAG:輪姦人数 = 人数
  TFLAG:輪姦現在順番 = 0

  FOR I, 0, 人数 - 1
    TFLAG:輪姦者_{I} = 加害者リスト:I
  NEXT

  ; 輪姦モード開始
  CALL GANGBANG_LOOP(被害者)
  RETURN

; 輪姦ループ
@GANGBANG_LOOP(被害者)
  WHILE TFLAG:輪姦中
    LOCAL 現在者 = TFLAG:輪姦者_{TFLAG:輪姦現在順番}

    ; 行為選択（AIまたはプレイヤー）
    IF 現在者 == PLAYER
      CALL GANGBANG_PLAYER_TURN(被害者)
    ELSE
      CALL GANGBANG_AI_TURN(被害者, 現在者)
    ENDIF

    ; 次の人へ
    TFLAG:輪姦現在順番 = (TFLAG:輪姦現在順番 + 1) % TFLAG:輪姦人数

    ; 終了判定
    CALL GANGBANG_CHECK_END(被害者)
  WEND
  RETURN

; 輪姦コマンド（新規番号帯）
; COM520-530: 輪姦専用

@COM520  ; 輪姦（モブ）
  ; モブ客システムと連携
  LOCAL 人数 = 3 + RAND:3  ; 3-5人

  ; モブ生成
  FOR I, 0, 人数 - 1
    CALL MOB_CLIENT_CREATE()
    TFLAG:輪姦者_{I} = RESULT
  NEXT

  CALL GANGBANG_START(TARGET, TFLAG:輪姦者, 人数)
  RETURN 1

@COM521  ; 輪姦（住人参加）
  ; 同室の住人を参加させる
  LOCAL 参加者:10
  LOCAL 参加数 = 0

  ; 同室住人を検索
  FOR キャラ, 0, CHARANUM - 1
    SIF キャラ == TARGET
      CONTINUE
    IF CFLAG:キャラ:現在位置 == CFLAG:TARGET:現在位置
      参加者:参加数 = キャラ
      参加数++
    ENDIF
  NEXT

  SIF 参加数 == 0
    PRINTL 他に誰もいない
    RETURN 0

  CALL GANGBANG_START(TARGET, 参加者, 参加数)
  RETURN 1
```
